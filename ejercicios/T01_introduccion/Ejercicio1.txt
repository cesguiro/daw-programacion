Algoritmo lavar_coche(encerado):
    Poner en marcha rodillos arrastre coche
    Encender rodillos enjabonado
    Empujar coche hasta zona enjabonado
    Repetir:
        Enjabonar coche
    Hasta: coche enjabonado
    Apagar rodillos enjabonado
    Encender rodillos aclarado
    Empujar coche hasta zona aclarado
    Repetir:
        Aclarar coche
    Hasta: coche aclarado
    Apagar rodillos aclarado
    Si (encerado):
        Empujar coche hasta zona encerado
        Repetir:
            Encerar coche
        Hasta: coche encerado
        Apagar rodillos encerado
    Empujar coche hasta zona secado
    Repetir:
        Secar coche
    Hasta: coche secado
    Empujar coche hasta salida
    Apagar rodillos arrastre coche
